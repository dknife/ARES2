\section{7차시 --- DC모터 입문: 회전과 룰렛}
\label{sec:ch07}

본 차시는 학습자가 처음으로 \emph{운동}을 코드로 제어하는 단계이다.
DC모터의 회전 방향(정/역)과 PWM 기반 속도 조절을 익히고, 모터에 원판을
달아 "룰렛 게임"을 완성한다. 출력 장치가 정적인 빛/소리에서 동적인
회전으로 확장되며, 학습자는 이제 "움직이는 작품"을 만들 수 있게 된다.

\subsection{미션 1 --- DC모터 기본 회전}
\label{subsec:ch07-m1}
\missionmeta{DC모터}{MOTOR}

\begin{storybox}
\sayares{알비! 이 모터에 전기를 공급하면 어떻게 돼?}
\sayalbi{삐릿- 코딩으로 전기를 보내면 빙글빙글 돌아요! 방향도 바꿀 수 있고 속도도 조절할 수 있어요.}
\end{storybox}

\subsubsection*{학습 목표}
\begin{itemize}[leftmargin=1.4em]
  \item DC모터 원리: 전기 → 자기장 → 회전 ($\bigstar$ 첫 등장)
  \item 정방향(앞)으로 3초 회전 후 정지
  \item 역방향(뒤)으로 3초 회전 후 정지
\end{itemize}

\begin{labbox}{샘플 코드 --- DC모터 기본 회전}
\begin{minted}{python}
# DC모터 기본 회전
# 정방향 전진
motor_forward(3)   # 3초 전진
motor_stop()
time.sleep(0.5)
# 역방향 후진
motor_backward(3)  # 3초 후진
motor_stop()
\end{minted}
\end{labbox}

\subsection{미션 2 --- 속도 조절 빠르게 느리게!}
\label{subsec:ch07-m2}
\missionmeta{DC모터}{MOTOR}

\begin{storybox}
\sayares{알비, DC모터로 전진·후진 속도를 조절해보자! 빠르게 갔다가 느리게 돌아오면 어떨까?}
\sayalbi{삐릿- 속도를 100\%에서 50\%로 바꾸면서 전진했다가 후진할 수 있어요! 두 가지 속도로 왕복해봐요!}
\end{storybox}

\subsubsection*{학습 목표}
\begin{itemize}[leftmargin=1.4em]
  \item 전진 최고속(100\%) 2초
  \item 전진 반속(50\%) 2초
  \item 정지!
  \item 후진 최고속(100\%) 2초
  \item 후진 반속(50\%) 2초 $\rightarrow$ 최종 정지
\end{itemize}

\begin{labbox}{샘플 코드 --- 전진·후진 속도 조절}
\begin{minted}{python}
# 전진·후진 속도 조절
motor_forward(2, 100)   # 전진 2초 최고속
motor_forward(2, 50)    # 전진 2초 반속
motor_stop()
motor_backward(2, 100)  # 후진 2초 최고속
motor_backward(2, 50)   # 후진 2초 반속
motor_stop()
\end{minted}
\end{labbox}

\subsection{미션 3 --- 차림표 원판 룰렛 돌리기!}
\label{subsec:ch07-m3}
\missionmeta{DC모터, 원판}{MOTOR}

\begin{storybox}
\sayares{이제 원판을 모터에 꽂고 화살표도 달았어! 돌리면 화살표가 어디에 멈추는지 봐야지!}
\sayalbi{삐릿- 룰렛이에요! 회전하다가 멈추면 그곳이 오늘의 선택!}
\end{storybox}

\subsubsection*{학습 목표}
\begin{itemize}[leftmargin=1.4em]
  \item 모터 최고 속도로 3초 회전 (룰렛 돌리기)
  \item 속도 100$\rightarrow$70$\rightarrow$40$\rightarrow$0\% 서서히 감속 후 정지
  \item 화살표가 멈추면 결과 확인
\end{itemize}

\begin{labbox}{샘플 코드 --- 룰렛 돌리기}
\begin{minted}{python}
# 룰렛 돌리기
motor_forward(3, 100)  # 최고 속도
time.sleep(3)
motor_forward(1, 70)   # 감속
time.sleep(1)
motor_forward(1, 40)   # 더 느리게
motor_stop()
\end{minted}
\end{labbox}

\subsection{미션 4 --- 나만의 화성 룰렛 게임!}
\label{subsec:ch07-m4}
\missionmeta{DC모터, 원판}{MOTOR}

\begin{storybox}
\sayares{알비! 원판 칸에 벌칙이나 상품 스티커를 붙이고 룰렛을 돌려봐! 멈추는 곳이 오늘의 미션!}
\sayalbi{삐릿♪ 빠르게 돌다가 서서히 멈추고, 마지막엔 살짝 후진해서 딱 멈추는 진짜 룰렛이에요!}
\end{storybox}

\subsubsection*{학습 목표}
\begin{itemize}[leftmargin=1.4em]
  \item 전진 3초 100\% (룰렛 최고속 회전)
  \item 전진 2초 50\% (서서히 감속)
  \item 멈춤!
  \item 후진 1초 50\% (미세 조정 $\rightarrow$ 정확히 멈춤)
\end{itemize}

\begin{labbox}{샘플 코드 --- 나만의 룰렛 게임}
\begin{minted}{python}
# 나만의 룰렛 게임
motor_forward(3, 100)   # 전진 3초 최고속
motor_forward(2, 50)    # 전진 2초 반속 감속
motor_stop()             # 멈춤
motor_backward(1, 50)   # 후진 1초 반속 (미세 조정)
motor_stop()             # 최종 정지
\end{minted}
\end{labbox}

\begin{caution}
DC모터는 큰 전류를 소모하므로 USB 전원만으로는 부족할 수 있다. 별도
배터리 팩 사용을 권장하며, 모터 정지 시에는 반드시 \texttt{motor\_stop()}을
호출해 전류 차단을 명시할 것.
\end{caution}
